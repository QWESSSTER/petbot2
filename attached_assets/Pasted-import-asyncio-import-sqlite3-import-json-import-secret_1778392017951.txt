def get_or_create_list(user_id: int, username: str) -> str:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT list_id FROM list_members WHERE user_id = ?”, (user_id,))
row = c.fetchone()
if row:
conn.close()
return row[0]
list_id = secrets.token_hex(4).upper()
c.execute(“INSERT INTO lists VALUES (?, ?)”, (list_id, user_id))
c.execute(“INSERT INTO list_members VALUES (?, ?, ?)”, (list_id, user_id, username))
conn.commit()
conn.close()
return list_id

def get_user_list_id(user_id: int) -> str | None:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT list_id FROM list_members WHERE user_id = ?”, (user_id,))
row = c.fetchone()
conn.close()
return row[0] if row else None

def get_locations(list_id: str) -> list:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT * FROM locations WHERE list_id = ?”, (list_id,))
rows = c.fetchall()
conn.close()
return rows

def add_location_db(list_id: str, user_id: int, username: str, data: dict):
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(”””
INSERT INTO locations (list_id, added_by, added_by_name, name, address, hours, avg_price, promotions, comment)
VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
“””, (
list_id, user_id, username,
data.get(“name”, “”),
data.get(“address”, “”),
data.get(“hours”, “”),
data.get(“avg_price”, “”),
data.get(“promotions”, “”),
data.get(“comment”, “”),
))
conn.commit()
conn.close()

def delete_location_db(loc_id: int):
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“DELETE FROM locations WHERE id = ?”, (loc_id,))
conn.commit()
conn.close()

def toggle_visited_db(loc_id: int) -> bool:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT visited FROM locations WHERE id = ?”, (loc_id,))
row = c.fetchone()
new_val = 0 if row[0] else 1
c.execute(“UPDATE locations SET visited = ? WHERE id = ?”, (new_val, loc_id))
conn.commit()
conn.close()
return bool(new_val)

def is_shared_list(list_id: str) -> bool:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT COUNT(*) FROM list_members WHERE list_id = ?”, (list_id,))
count = c.fetchone()[0]
conn.close()
return count > 1

def join_list_db(list_id: str, user_id: int, username: str) -> bool:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT 1 FROM lists WHERE list_id = ?”, (list_id,))
if not c.fetchone():
conn.close()
return False
c.execute(“DELETE FROM list_members WHERE user_id = ?”, (user_id,))
c.execute(“INSERT OR REPLACE INTO list_members VALUES (?, ?, ?)”, (list_id, user_id, username))
conn.commit()
conn.close()
return True

def get_list_members(list_id: str) -> list:
conn = sqlite3.connect(“locations.db”)
c = conn.cursor()
c.execute(“SELECT user_id FROM list_members WHERE list_id = ?”, (list_id,))
rows = c.fetchall()
conn.close()
return [r[0] for r in rows]

# ─── Форматирование ─────────────────────────────────────────────────────────────

def format_location(row, show_author=False) -> str:
# id, list_id, added_by, added_by_name, name, address, hours, avg_price, promotions, comment, visited
loc_id, _, _, added_by_name, name, address, hours, avg_price, promotions, comment, visited = row
parts = []
visited_mark = “ ✅” if visited else “”
if show_author:
parts.append(f”👤 *Добавил:* {added_by_name}”)
parts.append(f”📍 *{name}*{visited_mark}”)
if address:    parts.append(f”🗺 {address}”)
if hours:      parts.append(f”🕐 {hours}”)
if avg_price:  parts.append(f”💰 {avg_price}”)
if promotions: parts.append(f”🎁 {promotions}”)
if comment:    parts.append(f”💬 *{comment}*”)
return “\n”.join(parts)

async def extract_from_image(image_data: bytes) -> dict:
import PIL.Image
import io
image = PIL.Image.open(io.BytesIO(image_data))
prompt = (
“Извлеки информацию о заведении или локации из этого изображения. “
“Верни ТОЛЬКО валидный JSON без markdown, без пояснений:\n”
‘{“name”: “…”, “address”: “…”, “hours”: “…”, “avg_price”: “…”, “promotions”: “…”}\n’
“Если поле не найдено — ставь null.”
)
response = gemini_model.generate_content([prompt, image])
text = response.text.strip().replace(”`json", "").replace("`”, “”)
try:
return json.loads(text)
except Exception:
return {“name”: None, “address”: None, “hours”: None, “avg_price”: None, “promotions”: None}

# Ввод числа для перехода к конкретному месту
if current is None and message.text.strip().isdigit():
    list_id = get_user_list_id(user_id)
    if not list_id:
        return
    locations = get_locations(list_id)
    index = int(message.text.strip()) - 1
    if 0 <= index < len(locations):
        row = locations[index]

        await message.answer(f"Нет места с таким номером. Всего в списке: {len(locations)}")
    return

# FSM: заполнение полей
field_map = {
    AddLocation.asking_name:       "name",
    AddLocation.asking_address:    "address",
    AddLocation.asking_hours:      "hours",
    AddLocation.asking_price:      "avg_price",
    AddLocation.asking_promotions: "promotions",
    AddLocation.asking_comment:    "comment",
}
for state_cls, field in field_map.items():
    if current == state_cls.state:
        data = await state.get_data()
        data["location"][field] = message.text.strip()
        await state.update_data(location=data["location"])
        await ask_next_missing(message, state)
        return

# Новое место по тексту
await state.update_data(
    location={"name": message.text.strip(), "address": None, "hours": None, "avg_price": None, "promotions": None, "comment": None},
    user_id=user_id,
    username=username
)
await ask_next_missing(message, state)

@dp.callback_query(F.data == “skip_field”)
async def cb_skip_field(callback: CallbackQuery, state: FSMContext):
current = await state.get_state()
field_map = {
AddLocation.asking_address.state:    “address”,
AddLocation.asking_hours.state:      “hours”,
AddLocation.asking_price.state:      “avg_price”,
AddLocation.asking_promotions.state: “promotions”,
AddLocation.asking_comment.state:    “comment”,
}
if current in field_map:
data = await state.get_data()
data[“location”][field_map[current]] = “”
await state.update_data(location=data[“location”])
await callback.answer()
await ask_next_missing(callback, state)

@dp.callback_query(F.data.startswith(“nav:”))

)
